// ── Sheet names ────────────────────────────────────────────── const SH = { STUDENTS: "นักเรียน", SCORES: "คะแนน", ASSIGNMENTS: "งาน", CRITERIA: "เกณฑ์", SETTINGS: "การตั้งค่า", RAW: "GradeFlowRaw", // JSON backup (fallback) }; // ════════════════════════════════════════════════════════════ // HTTP HANDLERS // ════════════════════════════════════════════════════════════ function doGet(e) { try { const action = (e && e.parameter && e.parameter.action) || "load"; if (action === "ping") return ok({ pong: true, time: new Date().toISOString() }); return ok(loadAll()); } catch (err) { return ok({ error: err.message }); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); saveAll(payload); return ok({ saved: true, time: new Date().toISOString() }); } catch (err) { return ok({ error: err.message }); } } function ok(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } // ════════════════════════════════════════════════════════════ // SAVE — รับ JSON จากเว็บแล้วเขียนลง Sheets // ════════════════════════════════════════════════════════════ function saveAll(data) { if (data.students) saveStudents(data.students); if (data.assignments) saveAssignments(data.assignments); if (data.criteria) saveCriteria(data.criteria); if (data.settings) saveSettings_(data.settings); // เก็บ JSON ดิบไว้ด้วยเผื่อ restore saveRaw(data); } // ── นักเรียน ───────────────────────────────────────────────── function saveStudents(students) { const sh = getOrCreate(SH.STUDENTS); sh.clearContents(); const header = ["รหัส","เลขที่","ชื่อ-สกุล","ชั้น","ห้อง","วิชา","graded","gradedDate"]; const rows = [header]; Object.values(students).forEach(s => { rows.push([ s.id || "", s.no || "", s.name || "", s.level || "", s.room || "", s.subject || "", s.graded ? "TRUE" : "FALSE", s.gradedDate || "", ]); }); sh.getRange(1, 1, rows.length, header.length).setValues(rows); formatHeader(sh, header.length); } // ── คะแนน (1 แถว = 1 นักเรียน × 1 งาน) ───────────────────── function saveScores(students, assignments) { const sh = getOrCreate(SH.SCORES); sh.clearContents(); // สร้าง header แบบ dynamic ตาม criteria ของแต่ละงาน const assignList = Object.values(assignments || {}); const header = ["รหัสนักเรียน","ชื่อ","ชั้น","ห้อง","วิชา","รหัสงาน","ชื่องาน","วันที่ตรวจ","คะแนนรวม","เปอร์เซ็นต์","เกรด"]; const rows = [header]; Object.values(students).forEach(s => { const aScores = s.assignmentScores || {}; Object.entries(aScores).forEach(([aId, as_]) => { const asgn = (assignments || {})[aId]; const total = as_.scores ? as_.scores.reduce((a, b) => a + b, 0) : 0; const maxTotal = asgn ? asgn.criteria.reduce((a, c) => a + c.max, 0) : 100; const pct = maxTotal > 0 ? Math.round(total / maxTotal * 100) : 0; const grade = pct >= 80 ? "A" : pct >= 70 ? "B" : pct >= 60 ? "C" : "F"; rows.push([ s.id, s.name, s.level || "", s.room || "", s.subject || "", aId, asgn ? asgn.name : "", as_.gradedDate || "", total, pct + "%", grade, ]); }); }); if (rows.length > 1) { sh.getRange(1, 1, rows.length, header.length).setValues(rows); formatHeader(sh, header.length); } } // ── งาน ────────────────────────────────────────────────────── function saveAssignments(assignments) { const sh = getOrCreate(SH.ASSIGNMENTS); sh.clearContents(); const header = ["รหัสงาน","ชื่องาน","คะแนนเต็ม","น้ำหนัก%","คำอธิบาย","เกณฑ์ (JSON)"]; const rows = [header]; Object.values(assignments).forEach(a => { rows.push([ a.id, a.name, a.maxScore || 100, a.weight || "", a.desc || "", JSON.stringify(a.criteria || []), ]); }); if (rows.length > 1) { sh.getRange(1, 1, rows.length, header.length).setValues(rows); formatHeader(sh, header.length); } } // ── เกณฑ์ (default) ────────────────────────────────────────── function saveCriteria(criteria) { const sh = getOrCreate(SH.CRITERIA); sh.clearContents(); const header = ["ลำดับ","ชื่อเกณฑ์","คะแนนเต็ม"]; const rows = [header]; criteria.forEach((c, i) => rows.push([i + 1, c.name, c.max])); sh.getRange(1, 1, rows.length, 3).setValues(rows); formatHeader(sh, 3); } // ── การตั้งค่า ──────────────────────────────────────────────── function saveSettings_(settings) { const sh = getOrCreate(SH.SETTINGS); sh.clearContents(); const rows = [ ["ครูผู้สอน", settings.teacher || ""], ["โรงเรียน", settings.school || ""], ["กลุ่มสาระ", settings.dept || ""], ["ภาคเรียน", settings.term || ""], ["GAS URL", settings.gasUrl || ""], ]; sh.getRange(1, 1, rows.length, 2).setValues(rows); sh.getRange(1, 1, rows.length, 1).setFontWeight("bold"); sh.setColumnWidth(1, 120); sh.setColumnWidth(2, 300); } // ── JSON raw backup ─────────────────────────────────────────── function saveRaw(data) { const sh = getOrCreate(SH.RAW); sh.clearContents(); sh.getRange(1, 1).setValue(JSON.stringify(data)); sh.getRange(1, 2).setValue(new Date().toISOString()); } // ════════════════════════════════════════════════════════════ // LOAD — อ่านจาก Sheets แล้วคืน JSON กลับให้เว็บ // ════════════════════════════════════════════════════════════ function loadAll() { // อ่านจาก RAW JSON ก่อน (เร็วและครบ) try { const rawSh = SS.getSheetByName(SH.RAW); if (rawSh) { const val = rawSh.getRange(1, 1).getValue(); const time = rawSh.getRange(1, 2).getValue(); if (val) { const data = JSON.parse(val); data.time = time ? new Date(time).toISOString() : new Date().toISOString(); return { data }; } } } catch (e) {} return { data: null }; } // ════════════════════════════════════════════════════════════ // HELPERS // ════════════════════════════════════════════════════════════ function getOrCreate(name) { let sh = SS.getSheetByName(name); if (!sh) sh = SS.insertSheet(name); return sh; } function formatHeader(sh, numCols) { const hdr = sh.getRange(1, 1, 1, numCols); hdr.setBackground("#4f46e5"); hdr.setFontColor("#ffffff"); hdr.setFontWeight("bold"); sh.setFrozenRows(1); // auto resize columns (ไม่ทำถ้ามีข้อมูลน้อย เพราะช้า) if (numCols <= 12) { for (let i = 1; i <= numCols; i++) sh.autoResizeColumn(i); } } // ════════════════════════════════════════════════════════════ // ทดสอบจาก Apps Script Editor // เรียก testSave() เพื่อดูว่าสร้าง Sheet ถูกต้องไหม // ════════════════════════════════════════════════════════════ function testSave() { const sample = { students: { "67001": { id:"67001", no:"1", name:"นายสมชาย ใจดี", level:"ม.5", room:"1", subject:"วิทยาการคำนวณ", graded:true, gradedDate:"2568-03-01", assignmentScores:{ "demo1":{ scores:[28,22,18,13,10], gradedDate:"2568-03-01" } } } }, assignments: { "demo1":{ id:"demo1", name:"งานชิ้นที่ 1", maxScore:100, weight:20, desc:"", criteria:[{name:"เนื้อหา",max:30},{name:"วิเคราะห์",max:25},{name:"นำเสนอ",max:20},{name:"สมบูรณ์",max:15},{name:"ตรงเวลา",max:10}] } }, criteria: [{name:"เนื้อหา",max:30},{name:"วิเคราะห์",max:25},{name:"นำเสนอ",max:20},{name:"สมบูรณ์",max:15},{name:"ตรงเวลา",max:10}], settings: { teacher:"ครูทดสอบ", school:"โรงเรียนทดสอบ", dept:"วิทย์", term:"1/2568" } }; saveAll(sample); Logger.log("testSave OK"); }
